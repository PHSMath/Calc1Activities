\subsection*{Part 1: Basic Chain Rule}

For each of these functions, identify an inner function $g$ and an outer function $f$ to write the function in the form $f(g(x))$. Determine $f'(x)$, $g'(x)$, and $f'(g(x))$ and then apply the Chain Rule to determine the derivative of the function. \\

\begin{tcolorbox}
\noindent
Example: $z(x) = \sqrt{e^x + x^2}$
\begin{itemize}
    \item The inner function is $g(x) = e^x + x^2$. 
    \item The outer function is $f(x) = \sqrt{x}$. 
    \item $f'(x) = \frac{1}{2}x^{-1/2}$ and $g'(x) = e^x + 2x$. 
    \item $f'(g(x)) = \frac{1}{2}(e^x + x^2)^{-1/2}$. 
    \item Therefore 
    $$z'(x) = f'(g(x)) \cdot g'(x) = \frac{1}{2}(e^x + x^2)^{-1/2} \cdot (e^x + 2x)$. 
\end{itemize}
\end{tcolorbox}

\begin{enumerate}
    \item $a(x) = (x^2 + 2x - \sqrt{x})^{10}$
    \item $b(x) = cos(x^4)$
    \item $c(x) = 2^{\sin(x)}$
    \item $d(x) = \sqrt{\tan(x)}$
    \item $h(x) = \cot^5 (x)$
    \item $j(x) = (\sec(x) + e^x)^9$
\end{enumerate}
